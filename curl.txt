# Users
  # Register a new user request
    curl -X POST https://api.greasemeter.live/v1/users/register \
    -H "Content-Type: application/json" \
    -d '{
      "email": "test@example.com",
      "password": "passwordpassword",
      "name": "test123"
    }' \
    -w "\nHTTP Status: %{http_code}\n"

  # Register a new user response (204)
    {null}

  # Login a returning user request
    curl -X POST https://api.greasemeter.live/v1/users/login \
    -H "Content-Type: application/json" \
    -d '{
      "password": "passwordpassword",
      "name": "gemstonecrab"
    }' \
    -w "\nHTTP Status: %{http_code}\n"

  # Login a returning user response (200)
    {"token": string}

  # Forgot password request
    curl -X POST https://api.greasemeter.live/v1/users/forgot-password \
    -H "Content-Type: application/json" \
    -d '{
      "email": "test@example.com"
    }' \
    -w "\nHTTP Status: %{http_code}\n"

  # Forgot password response (204)
    {null}

  # Reset password request
    curl -X POST https://api.greasemeter.live/v1/users/reset-password/{string} \
    -H "Content-Type: application/json" \
    -d '{
      "password": "passwordpassword"
    }' \
    -w "\nHTTP Status: %{http_code}\n"

  # Reset password response (204)
    {null}

  # Delete user request
    curl -X DELETE https://api.greasemeter.live/v1/users \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer (token goes here)" \
    -w "\nHTTP Status: %{http_code}\n"

  # Delete user response (204)
    {null}

# Reviews
  # Reviews for place request
    curl -X GET https://api.greasemeter.live/v1/reviews/places/{int}?page=int&limit=int \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Reviews for place response (200)
    {
      "data": [
        "id": int
	"name": string (represents username)
	"rating": int
	"text": string
	"time": time
      ],
      "more": bool
    }

  # Create review request
    curl -X POST https://api.greasemeter.live/v1/reviews/places/{int} \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer (token goes here)" \
    -d '{
      "rating": 5
      "text": "i like it"
    }' \
    -w "\nHTTP Status: %{http_code}\n"

  # Create review response (201)
    {
      "id": int
      "time": time
    }

  # Reviews for user request
    curl -X GET https://api.greasemeter.live/v1/reviews?page=int&limit=int \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer (token goes here)" \
    -w "\nHTTP Status: %{http_code}\n"

  # Reviews for user response (200)
    {
      "data": [
        "id": int
	"name": string (represents place name)
	"rating": int
	"text": string
	"time": time
      ],
      "more": bool
    }

  # Update review request
    curl -X PATCH https://api.greasemeter.live/v1/reviews/{int} \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer (token goes here)" \
    -d '{
      "rating": 1
      "text": "i dont like it"
    }' \

    {null}
